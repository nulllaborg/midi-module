71void PlayDrumNote(const uint8_t midi_note, const uint8_t note_velocity) {
72 if (note_velocity > 0) {
73 g_midi.NoteOn(kChannel, midi_note, note_velocity);
74 g_midi.NoteOff(kChannel, midi_note);

88 g_midi.SetReverberation(kChannel, EM_MIDI_REVERBERATION_PLATE, kReverberationVolume, kReverberationDelayFeedback);

92void loop() {
93 for (uint8_t tick_no = 0; tick_no < kTickSize; tick_no++) {
94 g_tempo = g_tempo + (random(kTempoRandomRange) - kTempoRandomOffset);
95 g_tempo = constrain(g_tempo, kMinTempo, kMaxTempo);
96 PlayDrumNote(EM_MIDI_PERCUSSION_TIMBRE_1_NOTE_C_2_KICK_DRUM_1, kBassDrumTick[tick_no]);
97 PlayDrumNote(EM_MIDI_PERCUSSION_TIMBRE_1_NOTE_D_2_SNARE_DRUM_1, kSnareDrumTick[tick_no]);
98 PlayDrumNote(EM_MIDI_PERCUSSION_TIMBRE_1_NOTE_F_SHARP_2_CLOSED_HI_HAT, kHiHatCloseTick[tick_no]);
99 PlayDrumNote(EM_MIDI_PERCUSSION_TIMBRE_1_NOTE_A_SHARP_2_OPEN_HI_HAT, kHiHatOpenTick[tick_no]);
100 PlayDrumNote(EM_MIDI_PERCUSSION_TIMBRE_1_NOTE_G_SHARP_2_PEDAL_HI_HAT, kHiHatPedalTick[tick_no]);
101 delay(g_tempo);
102 }
